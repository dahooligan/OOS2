\section{Vor- und Nachteile}
\begin{frame}
\begin{columns}
\begin{column}{.5\textwidth}
\begin{block}{Vorteile}
\setbeamertemplate{itemize items}[square]
\begin{itemize}
  \item Producer und Consumer sind entkoppelt
  \item Anzahl Producer und Consumer kann unterschiedlich sein
  \item Konsistenz der Daten ist Gewährleistet
\end{itemize}
\end{block}
\end{column}
\begin{column}{.5\textwidth}
\begin{block}{Nachteile}
\setbeamertemplate{itemize items}[square]
\begin{itemize}
  \item Unbekannte Ausführungsreihenfolge bei Freigabe von Element
\end{itemize}
\end{block}
\end{column}
\end{columns}
\end{frame}

\subsection{Guarded Suspension}
\begin{frame}

\begin{block}{Guarded Suspension}
\setbeamertemplate{itemize items}[square]
\begin{itemize}
\item Objekt muss zum Objektaufruf in definiertem Zustand sein
\item zugreifende Objekte müssen warten bis das Objekt in diesem Zustand ist
\end{itemize}
\end{block}
